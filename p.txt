#!/usr/bin/env bash
set -euo pipefail

HOSTNAME_FQDN="${1:?usage: bootstrap.sh <hostname>}"

# packages
apt-get update && apt-get -y upgrade
apt-get install -y nfs-common curl apt-transport-https ca-certificates open-iscsi

# hostname
hostnamectl set-hostname "${HOSTNAME_FQDN%%.*}"

# time
# systemctl enable --now chrony

# local dns
# add relevant entries to /etc/hosts for k3s nodes and nfs

# swap
swapoff -a
sed -i.bak '/\sswap\s/s/^/#/' /etc/fstab

# modules
printf 'overlay\nbr_netfilter\n' > /etc/modules-load.d/k8s.conf
modprobe overlay; modprobe br_netfilter

# sysctls
cat <<'EOF' | tee /etc/sysctl.d/99-k8s.conf
# --- Kubernetes networking ---
net.bridge.bridge-nf-call-iptables  = 1
net.bridge.bridge-nf-call-ip6tables = 1
net.ipv4.ip_forward                 = 1

# --- Elasticsearch (required on any node that may run ES pods) ---
vm.max_map_count = 262144

# --- Many pods / watches (ES, Prometheus, operators) ---
fs.inotify.max_user_instances = 8192
fs.inotify.max_user_watches   = 524288
EOF

sysctl --system

# firewall
ufw allow from <k3s-subnet> to any port 6443 proto tcp comment 'Kubernetes API'
ufw allow from <k3s-subnet> to any port 2379:2380 proto tcp comment 'etcd server-to-server traffic'
ufw allow from <k3s-subnet> to any port 10250 proto tcp comment 'kubelet metrics'
ufw allow from <k3s-subnet> to any port 8472 proto udp comment 'Flannel VXLAN (prod)'
ufw allow from <k3s-subnet> to any port 7946 comment 'MetalLB speaker - udp/tcp'
ufw allow from <k3s-subnet> to any port 9200 proto tcp comment 'Elasticsearch API'

# 8. verify NFS
showmount -e nfs.example.com || { echo "NFS not reachable"; exit 1; }

echo "Base OS prep complete for ${HOSTNAME_FQDN}."

# apply..

The doc 03 components were installed by hand with helm/kubectl, so the first sync must adopt them without churn. The verified-safe procedure:

    Write the Application with syncPolicy: {} (no automation). On creation it reports OutOfSync / Healthy — it sees the live objects but won't touch them.
    argocd app diff <name> --core — confirm the only diff is Argo CD adding its argocd.argoproj.io/tracking-id annotations (no spec changes, no replacements). For MetalLB/ES this was exactly the case; nothing restarted.
    argocd app sync <name> --core once, then verify pods/LB IPs/ES health unchanged.
    Only then commit syncPolicy.automated: { prune: true, selfHeal: true } and push; root-platform rolls the automation out. (MetalLB chart sync used syncOptions: [ServerSideApply=true] to avoid the last-applied-annotation size limit on the CRDs.)

The MetalLB pool CRs need the controller's webhook up first (doc 03 §2). Use Argo CD sync-waves so the controller installs before the pool:

# platform/metallb/application.yaml  (the chart) — wave 0
metadata:
  annotations: { argocd.argoproj.io/sync-wave: "0" }

# platform/metallb/pool.yaml (IPAddressPool + L2Advertisement) — wave 1
metadata:
  annotations: { argocd.argoproj.io/sync-wave: "1" }

Container Registry — enabled on the GitLab instance (GitLab 19.0.2 EE) and on the projecta project. Endpoint is <gitlab>:5050 over plain HTTP. Probe it with curl -I http://<gitlab>:5050/v2/ (expect 401 = registry alive).
A GitLab Runner — installed on the GitLab server itself (apt install gitlab-runner, Ubuntu 24.04), docker executor, gitlab-runner user added to the docker group. GitLab 19 removed the registration-token flow: create the runner in the UI (project → Settings → CI/CD → Runners → New project runner, tick "Run untagged jobs") to mint a glrt-… authentication token, then gitlab-runner register --non-interactive --url http://<gitlab> --token glrt-… --executor docker --docker-image alpine:latest.
Node trust for the HTTP registry — containerd refuses plain-HTTP registries by default. On every node, write /etc/rancher/k3s/registries.yaml:

mirrors:
  "<gitlab>:5050":
    endpoint:
      - "http://<gitlab>:5050"
configs:
  "<gitlab>:5050":
    tls:
      insecure_skip_verify: true

then restart k3s (systemctl restart k3s on the server, k3s-agent on agents) — it re-renders certs.d/1<gitlab>:5050/hosts.toml. The restart is rolling and the control-plane VIP rides through it. (Agent nodes may not have /etc/rancher/k3s/ yet — mkdir -p first.)

    - changes: [ "projecta/**/*" ]

build-projectb:
  <<: *build
  variables: { APP: projectb }
  rules:
    - changes: [ "projectb/**/*" ]

deploy-staging:
  stage: deploy
  image:
    name: alpine/git
    entrypoint: [""]          # alpine/git's ENTRYPOINT is `git`; clear it or the
                              # job runs `git sh -c …` → "git: 'sh' is not a git command"
  variables:
    GIT_STRATEGY: none        # we only touch the GitOps/tags repo, not this checkout
  script:
    # One block scalar: colons/quotes inside stay literal, so a commit message like
    # "projecta: staging" can't be misparsed as a YAML mapping (see GOTCHA below).
    - |
      set -e

      cd gitops/source/$APP/staging
      yq -i '.image.tag = "'"$CI_COMMIT_SHORT_SHA"'"' image-tag.yaml
      git commit -am "$APP: staging -> $CI_COMMIT_SHORT_SHA [skip ci]"
      git push
  # Argo CD sees the commit and deploys to projecta-staging automatically.
  rules:
    - if: '$CI_COMMIT_BRANCH == "main"'

  Step 1 — Install cert-manager (with its CRDs)

  helm repo add jetstack https://charts.jetstack.io && helm repo update
  helm install cert-manager jetstack/cert-manager \
    --namespace cert-manager --create-namespace \
    --set crds.enabled=true

  Gate: kubectl get pods -n cert-manager → all 3 (cert-manager, cainjector, webhook) Running.

  Step 2 — Install the ADCS issuer (no simulator)

  Your earlier adcs-sim-* objects came from the demo/simulator. Check the values and make sure it's off before installing:

  helm repo add djkormo-adcs-issuer https://djkormo.github.io/adcs-issuer/ && helm repo update
  helm show values djkormo-adcs-issuer/adcs-issuer | grep -iA2 'simulator\|example'   # confirm any sim/example toggle is false
  helm install adcs-issuer djkormo-adcs-issuer/adcs-issuer \
    --namespace cert-manager

  Gate: kubectl get pods -n cert-manager | grep adcs → adcs-issuer-controller-manager Running 1/1.

  Now find the namespace the controller expects cluster-issuer secrets in — this is the #1 gotcha for a ClusterAdcsIssuer:

  kubectl get deploy adcs-issuer-controller-manager -n cert-manager -o yaml | grep -i cluster-resource-namespace

  Whatever that prints (call it $CRN — often adcs-issuer, but it depends on how the chart was installed) is where the credentials Secret must live. For a ClusterAdcsIssuer, the secret is NOT looked up next to your Certificate — it's
  looked up in $CRN. Use that namespace in Step 3.

  Step 3 — Credentials Secret (in $CRN from above)

  The secret must be Opaque with exactly username and password keys. Username is the NTLM account in DOMAIN\user form:

  kubectl create secret generic adcs-issuer-credentials \
    -n <CRN-from-step-2> \
    --from-literal=username='YOURDOMAIN\svc-certmgr' \
    --from-literal=password='YOUR-PASSWORD'

  Step 4 — Encode your CA bundle

  caBundle is the base64 of the CA chain (PEM) that signed your ADCS web-enrolment endpoint's HTTPS cert — so the issuer can trust the TLS connection to /certsrv. Usually your internal root + issuing CA concatenated:

  base64 -w0 ca-chain.pem    # copy this single-line output into caBundle below

  Step 5 — Create the ClusterAdcsIssuer

  # adcs-issuer.yaml
  apiVersion: adcs.certmanager.csf.nokia.com/v1
  kind: ClusterAdcsIssuer
  metadata:
    name: adcs
  spec:
    url: https://adcs.yourdomain.internal/certsrv   # ADCS web-enrolment endpoint (note /certsrv)
    templateName: WebServer                          # YOUR template name exactly
    caBundle: <paste base64 from step 4>
    credentialsRef:
      name: adcs-issuer-credentials                  # found in $CRN automatically
    statusCheckInterval: 6h
    retryInterval: 1h

  kubectl apply -f adcs-issuer.yaml

  Gate — this is the step you skipped before:
  kubectl get clusteradcsissuer adcs            # must actually appear now
  kubectl describe clusteradcsissuer adcs       # look for Ready/processing, no auth or secret-not-found errors
  If it complains it can't find the secret → the secret is in the wrong namespace (re-check $CRN). If it errors on auth/TLS → username format or caBundle is wrong. Fix this before issuing any cert.

  Step 6 — Test Certificate (note the group!)

  This is the field that was missing in our docs and causes "cannot find issuer" — for an external issuer issuerRef.group is mandatory:

  # test-cert.yaml
  apiVersion: cert-manager.io/v1
  kind: Certificate
  metadata:
    name: adcs-test
    namespace: default
  spec:
    secretName: adcs-test-tls
    commonName: adcs-test.yourdomain.internal
    dnsNames: ["adcs-test.yourdomain.internal"]
    issuerRef:
      name: adcs
      kind: ClusterAdcsIssuer
      group: adcs.certmanager.csf.nokia.com    # ← REQUIRED for the external issuer

  kubectl apply -f test-cert.yaml
  kubectl get certificate,certificaterequest,adcsrequest -n default
  kubectl describe certificate adcs-test -n default     # Ready: True
  kubectl logs -n cert-manager deploy/adcs-issuer-controller-manager --tail=30

  - "Couldn't get issuer ... not found" → group/name/kind mismatch (recheck Step 6) or issuer not Ready (back to Step 5).
  - "Processing ADCS request, Pending" that never clears → issuer resolved fine; now it's an ADCS-side problem (template name, the service account's enrol permission on that template, or NTLM auth). The controller logs will name it.

  Step 7 — Using it from Ingress

  For an external issuer the bare cert-manager.io/cluster-issuer annotation isn't enough; you must also give the group and kind:

  annotations:
    cert-manager.io/cluster-issuer: adcs
    cert-manager.io/issuer-kind:  ClusterAdcsIssuer
    cert-manager.io/issuer-group: adcs.certmanager.csf.nokia.com
